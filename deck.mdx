## Destrukturering
```typescript
const [a, b] = [1, 2];
console.log(a); // 1
console.log(b); // 2

const [first, ...rest] = ["a", "b", "c", "d"];
console.log(first); // "a"
console.log(rest);  // ["b", "c", "d"]
```

---

## Javascript-objekter
```typescript
const person = {
  name: "Magnus",
  age: 29,
  makeSound: () => console.log("Jasså!")
};

console.log(person.name);    // "Magnus"
console.log(person["name"]); // "Magnus"
```

---

## Destrukturering av objekter
```typescript
const { message, ...otherStuff } = {
  date: "21.04.21",
  message: "Velkommen til workshop!"
};

console.log(message);    // "Velkommen til workshop!"
console.log(otherStuff); // { date: "21.04.21" }
```

---

## Spread
```typescript
const numbers = [1, 2, 3];
const moreNumbers = [0, ...numbers, 4];
// [0, 1, 2, 3, 4]

console.log(moreNumbers); // [0, 1, 2, 3, 4]
```

---

## Spread
```typescript
const something = {
  x: 3,
  y: "Hei",
};
const other = {
  y: "Hallo",
  z: [1, 2],
};
const merged = { ...something, ...other };
console.log(merged);
// {
//   x: 3,
//   y: "Hallo",
//   z: [1, 2]
// }

// { x: 3, y: "Hei", y: "Hallo", z: [1, 2] }
```

## Typer og inferens

```typescript
const x: boolean = true;

let z = 5;
z = "Jeg vil være en streng!";
// Type 'string' is not assignable to type 'number'.

let w: number | string = 5;
w = "Nå er det ok!";
```

## Typedeklarasjoner
```typescript
type Foo = {
  bar: number[],
  baz: string
}

const foo: Foo = { bar: [9], baz: "Hei" };

const foo2: Foo = { bar: [9], baz: 1 }
// Type 'number' is not assignable to type 'string'.
```
