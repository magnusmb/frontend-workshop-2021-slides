## Typeinferens

```typescript
const x: number = 1;

const y: number = "Hei";
// Type 'string' is not assignable to type 'number'.

let z = 5;
z = "Jeg vil være en streng!";
// Type 'string' is not assignable to type 'number'.

let w: number | string = 5;
w = "Nå er det ok!";
```

## Typedeklarasjoner
```typescript
type Foo = {
  bar: number[],
  baz: "Hei"
}

const foo: Foo = { bar: [9], baz: "Hallo" };
// Type '"Hallo"' is not assignable to type '"Hei"'.
```

## Funksjoner
```typescript
function add(a: number, b:number) {
  return a + b;
}

const add = (a: number, b: number) => {
  return a + b;
};
```

## Destrukturering
```typescript
const [a, b] = [1, 2];
console.log(a); // 1
console.log(b); // 2

const [foo, ...bar] = ["a", "b", "c", "d"];
console.log(foo); // "a"
console.log(bar); // ["b", "c", "d"]
```

---

## Med objekter
```typescript
const { y, ...z } = {
  x: 30,
  y: "Velkommen til workshop!"
};

console.log(y); // "Velkommen til workshop!"
console.log(z); // { x: 30 }
```

---

## Spread
```typescript
const foo = {
  x: 1,
  y: "Hei"
};
const bar = {
  y: "Hallo",
  z: [1, 2]
};
const baz = { ...foo, ...bar };
console.log(baz);
// {
//   x: 1,
//   y: "Hallo",
//   z: [1, 2]
// }
```
